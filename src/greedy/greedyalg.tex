\begin{frame}{Greedy Algorithm for Pair Packing}


\begin{algorithm}[H]
\begin{algorithmic}[1]

\STATE $I \gets \emptyset$
\WHILE{$I$ is not maximal}
\STATE $c_{min} \gets \argmin_{\{c : I \cup \{c\} \text{ is independent }\}}|c|$
\STATE $I \gets I \cup \{ c_{min} \}$
\ENDWHILE
\RETURN $I$


\end{algorithmic}
\caption{Greedy Algorithm}
\end{algorithm}


\end{frame}

\begin{frame}{First Bound}

Assume that every \pack{} in $\mathcal{I}^*$ has more than two colors


\pause
\begin{observation}
$$ |\mathcal{I}^*| \leq \frac{|P|}{2} $$
\end{observation}

\pause
\begin{corollary}
$$
r 													:= 
\frac{\alpha |P| - |\mathcal{I}^*|}{\alpha (|P| - |\mathcal{I}^*|)} 	\leq 
2 - \frac{1}{\alpha} 								\leq 
2
$$
\end{corollary}


\end{frame}

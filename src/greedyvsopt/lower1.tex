\begin{frame}{First Bound}

Assume that every \pack{} in $I^*$ has more than two colors


\pause
\begin{observation}
$$ |I^*| \leq \frac{|P|}{2} $$
\end{observation}

\pause
\begin{corollary}
$$
r 													:= 
\frac{\alpha |P| - |I^*|}{\alpha (|P| - |I^*|)} 	\leq 
2 - \frac{1}{\alpha} 								\leq 
2
$$
\end{corollary}


\end{frame}

\begin{frame}{Greedy vs Optimal}
Notation:
\begin{itemize}
\boldmath	
\pause\item
$I$: greedy packing

\pause\item
$I^*$: optimal packing

\pause\item
$\alpha := \frac{|I^*|}{|I|}$ 

\end{itemize}

\pause
\begin{observation}
The approximation ratio for 2-CR is
$$ r := \frac{|P| - |I|}{|P| - |I^*|} = \frac{|P| - |I|}{|P| - \alpha|I|}$$
\end{observation}

\begin{itemize}
\pause\item
We need to \textbf{upper bound} $r$
\pause\item
Can be done by \textbf{lower bounding} $|P|$
\end{itemize}


\end{frame}

\begin{frame}{Greedy Algorithm for Pair Packing}


\begin{algorithm}[H]
\begin{algorithmic}[1]

\STATE $I \gets \emptyset$
\WHILE{$I$ is not maximal}
\STATE $I \gets I \cup \{\text{smallest independent subgraph}\}$
\ENDWHILE
\RETURN $I$


\end{algorithmic}
\caption{Greedy Algorithm}
\end{algorithm}


\end{frame}

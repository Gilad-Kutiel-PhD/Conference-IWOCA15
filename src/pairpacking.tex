\item \textbf<1> 
{Given a 2-CR instance}

\item \textbf<2>
{A packing is a set of connected subgraphs, s.t.:}
	
	\begin{itemize}
	
	\item \textbf<3> 
	{Every subgraph contains a pair}
	
	\item \textbf<4>
	{No two subgraphs share a common color} 
	
	\end{itemize} 

\item \textbf<5>
{The goal is to find a maximum packing}

\end{itemize}
